use cs332;

drop table if exists cars;

create table cars (
   car_id INT AUTO_INCREMENT PRIMARY KEY,
   model VARCHAR(20),
   make VARCHAR(20),
   year YEAR,
   seats INT,
   mileage INT,
   rating FLOAT(2,1),
   description VARCHAR(20)
);
